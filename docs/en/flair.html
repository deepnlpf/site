Python
Output JSON
Output XML
from deepnlpf.pipeline import Pipeline

path_pipeline = '<path>/pipeline_flair.json'
sentence = 'The grass is green.'

nlp = Pipeline(_input=sentence, pipeline=path_pipeline)
annotation = nlp.annotate()
